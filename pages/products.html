// js/pages/products.js document.addEventListener('DOMContentLoaded', () => { const container = document.getElementById('all-products'); if (!container) return; // Función para renderizar productos function renderProducts(products) { if (!products || products.length === 0) { container.innerHTML = '

No hay productos disponibles en este momento.

'; return; } container.innerHTML = products.map(p => `

${p.name}

${p.description}

$${p.price} CUP Ver
`).join(''); } // Esperar a que CONFIG esté listo (por si acaso) if (window.CONFIG && window.CONFIG.products) { // Mostrar todos los productos (puedes filtrar por inStock si quieres: .filter(p => p.inStock)) renderProducts(window.CONFIG.products); } else { // Si tarda, reintentar const checkInterval = setInterval(() => { if (window.CONFIG && window.CONFIG.products) { clearInterval(checkInterval); renderProducts(window.CONFIG.products); } }, 100); // Timeout de seguridad después de 5 segundos setTimeout(() => clearInterval(checkInterval), 5000); } });
